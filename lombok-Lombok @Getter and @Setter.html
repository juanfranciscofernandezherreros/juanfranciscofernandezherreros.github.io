Use @Getter and @Setter on Class @Getter @Setter public class Account { private String username; private String password; } public class Account { private String username; private String password; public String getUsername() { return this.username; } public String getPassword() { return this.password; } public void setUsername(String username) { this.username = username; } public void setPassword(String password) { this.password = password; } } 3. Use @Getter and @Setter on Fields public class Account { @Getter @Setter private String username; @Getter @Setter private String password; } public class Account { private String username; private String password; public String getUsername() { return this.username; } public String getPassword() { return this.password; } public void setUsername(String username) { this.username = username; } public void setPassword(String password) { this.password = password; } }
